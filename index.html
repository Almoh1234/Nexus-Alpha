        async function fetchGemini(prompt, systemPrompt = "You are Nexus Terminal AI.") {

            const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;

            const payload = {

                contents: [{ parts: [{ text: prompt }] }],

                systemInstruction: { parts: [{ text: systemPrompt }] }

            };

            let retries = 0;

            const delays = [1000, 2000, 4000, 8000, 16000];

            while (retries < 5) {

                try {

                    const res = await fetch(url, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(payload) });

                    const data = await res.json();

                    return data.candidates?.[0]?.content?.parts?.[0]?.text || "No data received.";

                } catch (e) {

                    await new Promise(r => setTimeout(r, delays[retries]));

                    retries++;

                }

            }

            return "Connection Failure.";

        }

        function playPCM(base64) {

            const binary = atob(base64);

            const bytes = new Uint8Array(binary.length);

            for (let i = 0; i < binary.length; i++) bytes[i] = binary.charCodeAt(i);

            const buffer = new ArrayBuffer(44 + bytes.length);

            const view = new DataView(buffer);

            const sampleRate = 24000;

            const writeString = (v, o, s) => { for (let i = 0; i < s.length; i++) v.setUint8(o + i, s.charCodeAt(i)); };

            writeString(view, 0, 'RIFF'); view.setUint32(4, 32 + bytes.length, true);

            writeString(view, 8, 'WAVE'); writeString(view, 12, 'fmt ');

            view.setUint32(16, 16, true); view.setUint16(20, 1, true); view.setUint16(22, 1, true);

            view.setUint32(24, sampleRate, true); view.setUint32(28, sampleRate * 2, true);

            view.setUint16(32, 2, true); view.setUint16(34, 16, true);

            writeString(view, 36, 'data'); view.setUint32(40, bytes.length, true);

            new Uint8Array(buffer, 44).set(bytes);

            new Audio(URL.createObjectURL(new Blob([buffer], { type: 'audio/wav' }))).play();

        }

        function updateSignalWatch(headlines) {

            const signals = {

                trump: { keywords: ['trump', 'republican', 'election', 'potus', 'djt'], elId: 'signal-trump' },

                musk: { keywords: ['musk', 'elon', 'tesla', 'spacex', 'x.com', 'starlink'], elId: 'signal-musk' },

                fed: { keywords: ['fed', 'powell', 'interest rate', 'fomc', 'inflation', 'beige book'], elId: 'signal-fed' }

            };


            Object.keys(signals).forEach(key => {

                const config = signals[key];

                const triggerHeadline = headlines.find(h => config.keywords.some(k => h.toLowerCase().includes(k.toLowerCase())));

                

                const el = document.getElementById(config.elId);

                const statusLabel = el.querySelector('.signal-status');

                const headlineEl = el.querySelector('.trigger-headline');


                if (triggerHeadline) {

                    statusLabel.innerText = 'ACTIVE';

                    statusLabel.className = 'signal-status text-[8px] signal-active font-bold uppercase animate-pulse';

                    headlineEl.innerHTML = `<span class="trigger-text"><i class="fas fa-satellite-dish mr-1"></i> ${triggerHeadline.substring(0, 60)}...</span>`;

                } else {

                    statusLabel.innerText = 'Idle';

                    statusLabel.className = 'signal-status text-[8px] signal-idle font-bold uppercase';

                    headlineEl.innerHTML = '';

                }

            });

        }

        async function fetchNews() {

            try {

                const res = await fetch(`https://finnhub.io/api/v1/news?category=general&token=${FINNHUB_KEY}`);

                let data = await res.json();

                

                // Sort newest to oldest based on timestamp

                data.sort((a, b) => b.datetime - a.datetime);


                const container = document.getElementById('news-grid');

                const breakingContainer = document.getElementById('breaking-grid');

                const loader = document.getElementById('news-loader');

                

                if (loader) loader.remove();


                const recentNews = data.slice(0, 30); // Grab a larger pool to filter

                currentHeadlines = recentNews.map(n => n.headline);

                updateSignalWatch(currentHeadlines);


                // Reverse to process newest last (so prepending puts newest on top)

                for (const item of [...recentNews].reverse()) {

                    if (seenNewsIds.has(item.id)) continue;

                    seenNewsIds.add(item.id);


                    const sourceName = (item.source || "FINNHUB").toUpperCase();

                    

                    // Priority check for requested networks

                    const isPriorityNetwork = PRIORITY_NETWORKS.some(net => 

                        sourceName.includes(net.toUpperCase()) || 

                        item.headline.toUpperCase().includes(net.toUpperCase())

                    );


                    const impact = await analyzeNewsImpact(item.headline);

                    const isBreaking = impact.includes("MOVING") || isPriorityNetwork;

                    const impactClass = isPriorityNetwork ? "badge-priority" : (isBreaking ? "badge-market-moving" : "badge-neutral");

                    const statusText = isPriorityNetwork ? "NETWORK INTEL" : impact;

                    

                    const card = document.createElement('div');

                    card.id = `intel-${item.id}`;

                    card.className = `news-item p-4 bg-[#050505] border border-gray-900 rounded ${isBreaking ? 'breaking-item' : ''}`;

                    card.innerHTML = `

                        <div class="flex justify-between items-start mb-2">

                            <div class="flex items-center gap-2">

                                <span class="text-[8px] font-black uppercase text-blue-500 tracking-widest bg-blue-500/10 px-2 py-0.5 rounded border border-blue-500/20">SRC: ${sourceName}</span>

                                <span class="text-[8px] font-bold uppercase px-2 py-0.5 rounded ${impactClass}">${statusText}</span>

                            </div>

                            <button onclick="dismissIntel('${item.id}')" class="dismiss-btn text-gray-600 hover:text-green-500 transition-colors p-1" title="Mark as Read">

                                <i class="fas fa-check-circle text-xs"></i>

                            </button>

                        </div>

                        <h3 class="${isBreaking ? 'text-[12px]' : 'text-[11px]'} font-bold text-gray-100 leading-tight">${item.headline}</h3>

                        <div class="flex justify-between items-center mt-3">

                            <p class="text-[8px] text-gray-600 font-bold uppercase"><i class="far fa-clock mr-1"></i> ${new Date(item.datetime * 1000).toLocaleTimeString()}</p>

                            <a href="${item.url}" target="_blank" class="text-[8px] text-blue-400 hover:text-white transition-colors font-bold uppercase tracking-tighter">Source</a>

                        </div>`;


                    if (isBreaking) {

                        breakingContainer.prepend(card);

                    } else {

                        container.prepend(card);

                    }

                }

            } catch (e) { console.error("Intel fetch error:", e); }

        }

        function initWebSocket() {

            const socket = new WebSocket(`wss://ws.finnhub.io?token=${FINNHUB_KEY}`);

            socket.addEventListener('open', () => {

                document.getElementById('ws-status-dot').className = 'text-green-500';

                ['SPY', 'BINANCE:BTCUSDT', 'BINANCE:ETHUSDT'].forEach(s => socket.send(JSON.stringify({'type':'subscribe', 'symbol': s})));

            });

            socket.addEventListener('message', (e) => {

                const msg = JSON.parse(e.data);

                if (msg.type === 'trade') {

                    msg.data.forEach(trade => {

                        const el = document.getElementById(`price-${trade.s}`);

                        if (el) {

                            const lp = lastPrices[trade.s] || 0;

                            el.innerText = `$${trade.p.toLocaleString(undefined, {minimumFractionDigits: 2})}`;

                            el.classList.remove('price-up', 'price-down');

                            void el.offsetWidth;

                            el.classList.add(trade.p > lp ? 'price-up' : 'price-down');

                            lastPrices[trade.s] = trade.p;

                        }

                    });

                }

            });

        }
